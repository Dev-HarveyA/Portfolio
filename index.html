[RequireComponent(typeof(Rigidbody))]
public sealed class PlayerCharacterController : MonoBehaviour
{
    [Header("Movement")]
    [SerializeField] private float moveSpeed = 8f;
    [SerializeField] private float acceleration = 20f;
    [SerializeField] private float deceleration = 25f;
    [SerializeField] private float airControlMultiplier = 0.4f;

    [Header("Jumping")]
    [SerializeField] private float jumpForce = 10f;
    [SerializeField] private float gravityScale = 2.5f;
    [SerializeField] private float fallGravityScale = 4f;
    [SerializeField] private float coyoteTime = 0.15f;
    [SerializeField] private float jumpBufferTime = 0.12f;
    [SerializeField] private int maxExtraJumps = 1;

    [Header("Ground Detection")]
    [SerializeField] private LayerMask groundMask;
    [SerializeField] private float groundCheckRadius = 0.25f;
    [SerializeField] private Transform groundCheckPoint;

    private Rigidbody rb;
    private Vector3 moveInput;
    private float coyoteTimer;
    private float jumpBufferTimer;
    private int extraJumpsLeft;
    private bool isGrounded;
    private bool wasGrounded;
    private bool jumpConsumed;

    private static readonly float GravityBase = Physics.gravity.y;

    private void Awake()
    {
        rb = GetComponent<Rigidbody>();
        rb.freezeRotation = true;
        rb.interpolation = RigidbodyInterpolation.Interpolate;
    }

    private void Update()
    {
        ReadInput();
        UpdateGroundState();
        HandleCoyoteTime();
        HandleJumpBuffer();
        TryJump();
    }

    private void FixedUpdate()
    {
        ApplyMovement();
        ApplyCustomGravity();
    }

    private void ReadInput()
    {
        float h = Input.GetAxisRaw("Horizontal");
        float v = Input.GetAxisRaw("Vertical");
        moveInput = new Vector3(h, 0f, v).normalized;
        if (Input.GetButtonDown("Jump"))
            jumpBufferTimer = jumpBufferTime;
    }

    private void UpdateGroundState()
    {
        wasGrounded = isGrounded;
        isGrounded = Physics.CheckSphere(
            groundCheckPoint.position, groundCheckRadius,
            groundMask, QueryTriggerInteraction.Ignore);
        if (isGrounded && !wasGrounded) OnLanded();
    }

    private void OnLanded()
    {
        extraJumpsLeft = maxExtraJumps;
        jumpConsumed = false;
        StopCoroutine(nameof(LandingSquishCoroutine));
        StartCoroutine(nameof(LandingSquishCoroutine));
    }

    private IEnumerator LandingSquishCoroutine()
    {
        Vector3 squish = new Vector3(1.15f, 0.8f, 1.15f);
        float t = 0f;
        while (t < 1f)
        {
            transform.localScale = Vector3.Lerp(squish, Vector3.one, t);
            t += Time.deltaTime * 12f;
            yield return null;
        }
        transform.localScale = Vector3.one;
    }

    private void HandleCoyoteTime()
    {
        if (isGrounded) coyoteTimer = coyoteTime;
        else coyoteTimer -= Time.deltaTime;
    }

    private void HandleJumpBuffer()
    {
        if (jumpBufferTimer > 0f) jumpBufferTimer -= Time.deltaTime;
    }

    private void TryJump()
    {
        bool canCoyoteJump = coyoteTimer > 0f && !jumpConsumed;
        bool hasBuffer = jumpBufferTimer > 0f;
        if (!hasBuffer) return;
        if (canCoyoteJump)
        {
            PerformJump(jumpForce);
            jumpConsumed = true;
            jumpBufferTimer = 0f;
            coyoteTimer = 0f;
        }
        else if (extraJumpsLeft > 0 && !isGrounded)
        {
            PerformJump(jumpForce * 0.85f);
            extraJumpsLeft--;
            jumpBufferTimer = 0f;
        }
    }

    private void PerformJump(float force)
    {
        Vector3 v = rb.velocity;
        v.y = 0f;
        rb.velocity = v;
        rb.AddForce(Vector3.up * force, ForceMode.Impulse);
    }

    private void ApplyMovement()
    {
        Vector3 target = moveInput * moveSpeed;
        Vector3 current = new Vector3(rb.velocity.x, 0f, rb.velocity.z);
        float control = isGrounded ? 1f : airControlMultiplier;
        float rate = moveInput.sqrMagnitude > 0.01f
            ? acceleration * control
            : deceleration * control;
        Vector3 next = Vector3.MoveTowards(current, target, rate * Time.fixedDeltaTime);
        rb.velocity = new Vector3(next.x, rb.velocity.y, next.z);
    }

    private void ApplyCustomGravity()
    {
        float scale = rb.velocity.y < 0f ? fallGravityScale : gravityScale;
        rb.AddForce(Vector3.up * GravityBase * (scale - 1f), ForceMode.Acceleration);
    }

    private void OnDrawGizmosSelected()
    {
        if (groundCheckPoint == null) return;
        Gizmos.color = isGrounded ? Color.green : Color.red;
        Gizmos.DrawWireSphere(groundCheckPoint.position, groundCheckRadius);
    }
}

public sealed class ObjectPooler : MonoBehaviour
{
    [Serializable]
    public struct PoolDefinition
    {
        public string key;
        public GameObject prefab;
        public int initialSize;
        public bool expandable;
        public Transform parent;
    }

    [SerializeField] private PoolDefinition[] pools;

    private readonly Dictionary<string, Queue<GameObject>> available = new();
    private readonly Dictionary<string, List<GameObject>> allObjects = new();
    private readonly Dictionary<string, PoolDefinition> definitions = new();

    public static ObjectPooler Instance { get; private set; }

    private void Awake()
    {
        if (Instance != null && Instance != this) { Destroy(gameObject); return; }
        Instance = this;
        DontDestroyOnLoad(gameObject);
        InitialisePools();
    }

    private void InitialisePools()
    {
        foreach (PoolDefinition def in pools)
        {
            if (available.ContainsKey(def.key)) continue;
            available[def.key] = new Queue<GameObject>();
            allObjects[def.key] = new List<GameObject>(def.initialSize);
            definitions[def.key] = def;
            for (int i = 0; i < def.initialSize; i++)
                CreateAndEnqueue(def);
        }
    }

    private GameObject CreateAndEnqueue(PoolDefinition def)
    {
        GameObject go = Instantiate(def.prefab, def.parent);
        go.SetActive(false);
        available[def.key].Enqueue(go);
        allObjects[def.key].Add(go);
        if (go.TryGetComponent(out IPoolable poolable)) poolable.OnCreated();
        return go;
    }

    public GameObject Get(string key, Vector3 position, Quaternion rotation)
    {
        if (!definitions.TryGetValue(key, out PoolDefinition def)) return null;
        if (available[key].Count == 0)
        {
            if (!def.expandable) return null;
            CreateAndEnqueue(def);
        }
        GameObject go = available[key].Dequeue();
        go.transform.SetPositionAndRotation(position, rotation);
        go.SetActive(true);
        if (go.TryGetComponent(out IPoolable poolable)) poolable.OnSpawned();
        return go;
    }

    public void Return(string key, GameObject go)
    {
        if (!available.ContainsKey(key)) { Destroy(go); return; }
        if (go.TryGetComponent(out IPoolable poolable)) poolable.OnDespawned();
        go.SetActive(false);
        available[key].Enqueue(go);
    }

    public void ReturnAll(string key)
    {
        if (!allObjects.TryGetValue(key, out List<GameObject> list)) return;
        foreach (GameObject go in list)
            if (go.activeSelf) Return(key, go);
    }

    public int GetActiveCount(string key)
    {
        if (!allObjects.TryGetValue(key, out List<GameObject> list)) return 0;
        int count = 0;
        foreach (GameObject go in list) if (go.activeSelf) count++;
        return count;
    }
}

    void OnSpawned();
    void OnDespawned();
}

using System;
using System.Collections.Generic;

public class StateMachine<TOwner>
{
    private readonly TOwner owner;
    private IState<TOwner> currentState;
    private IState<TOwner> previousState;
    private readonly Dictionary<Type, IState<TOwner>> states = new();
    private readonly List<Transition<TOwner>> anyTransitions = new();
    private readonly Dictionary<Type, List<Transition<TOwner>>> stateTransitions = new();

    public IState<TOwner> CurrentState => currentState;
    public event Action<IState<TOwner>, IState<TOwner>> OnStateChanged;

    public StateMachine(TOwner owner) { this.owner = owner; }

    public void AddState<TState>() where TState : IState<TOwner>, new()
    {
        TState state = new();
        state.Initialise(owner);
        states[typeof(TState)] = state;
    }

    public void AddTransition<TFrom, TTo>(Func<bool> condition)
        where TFrom : IState<TOwner>
        where TTo   : IState<TOwner>
    {
        Type from = typeof(TFrom);
        if (!stateTransitions.ContainsKey(from))
            stateTransitions[from] = new List<Transition<TOwner>>();
        stateTransitions[from].Add(new Transition<TOwner>(GetState<TTo>(), condition));
    }

    public void AddAnyTransition<TTo>(Func<bool> condition) where TTo : IState<TOwner>
    {
        anyTransitions.Add(new Transition<TOwner>(GetState<TTo>(), condition));
    }

    public void SetInitialState<TState>() where TState : IState<TOwner>
    {
        currentState = GetState<TState>();
        currentState.Enter();
    }

    public void Tick()
    {
        Transition<TOwner> triggered = GetTriggeredTransition();
        if (triggered != null) ChangeState(triggered.To);
        currentState?.Tick();
    }

    public void TickFixed() => currentState?.TickFixed();

    public void ForceState<TState>() where TState : IState<TOwner>
        => ChangeState(GetState<TState>());

    private Transition<TOwner> GetTriggeredTransition()
    {
        foreach (Transition<TOwner> t in anyTransitions)
            if (t.Condition()) return t;
        Type type = currentState?.GetType();
        if (type != null && stateTransitions.TryGetValue(type, out var list))
            foreach (Transition<TOwner> t in list)
                if (t.Condition()) return t;
        return null;
    }

    private void ChangeState(IState<TOwner> next)
    {
        if (next == currentState) return;
        previousState = currentState;
        currentState?.Exit();
        currentState = next;
        currentState.Enter();
        OnStateChanged?.Invoke(previousState, currentState);
    }

    private IState<TOwner> GetState<TState>() where TState : IState<TOwner>
    {
        states.TryGetValue(typeof(TState), out IState<TOwner> state);
        return state;
    }
}

public interface IState<TOwner>
{
    void Initialise(TOwner owner);
    void Enter();
    void Tick();
    void TickFixed();
    void Exit();
}

public abstract class BaseState<TOwner> : IState<TOwner>
{
    protected TOwner Owner { get; private set; }
    public void Initialise(TOwner owner) => Owner = owner;
    public virtual void Enter() { }
    public virtual void Tick() { }
    public virtual void TickFixed() { }
    public virtual void Exit() { }
}

internal sealed class Transition<TOwner>
{
    public readonly IState<TOwner> To;
    public readonly Func<bool> Condition;
    public Transition(IState<TOwner> to, Func<bool> condition) { To = to; Condition = condition; }
}

public class EnemyAI : MonoBehaviour
{
    [SerializeField] private float detectionRange = 10f;
    [SerializeField] private float attackRange = 2f;
    [SerializeField] private float moveSpeed = 4f;

    private StateMachine<EnemyAI> sm;
    private Transform target;
    public float DistanceToTarget => target != null
        ? Vector3.Distance(transform.position, target.position) : float.MaxValue;

    private void Awake()
    {
        target = GameObject.FindGameObjectWithTag("Player")?.transform;
        sm = new StateMachine<EnemyAI>(this);
        sm.AddState<EnemyIdleState>();
        sm.AddState<EnemyChaseState>();
        sm.AddState<EnemyAttackState>();
        sm.AddTransition<EnemyIdleState,   EnemyChaseState> (() => DistanceToTarget < detectionRange);
        sm.AddTransition<EnemyChaseState,  EnemyIdleState>  (() => DistanceToTarget > detectionRange * 1.2f);
        sm.AddTransition<EnemyChaseState,  EnemyAttackState>(() => DistanceToTarget < attackRange);
        sm.AddTransition<EnemyAttackState, EnemyChaseState> (() => DistanceToTarget >= attackRange);
        sm.SetInitialState<EnemyIdleState>();
    }

    private void Update()       => sm.Tick();
    private void FixedUpdate()  => sm.TickFixed();
    public void MoveTowardsTarget()
    {
        if (target == null) return;
        transform.position = Vector3.MoveTowards(
            transform.position, target.position, moveSpeed * Time.deltaTime);
    }
    public void TriggerAttack()  { }
    public void ResetCooldown()  { }
}

public static class EventBus<T> where T : struct
{
    private static readonly HashSet<IEventBinding<T>> bindings = new();

    public static void Register(IEventBinding<T> binding)   => bindings.Add(binding);
    public static void Deregister(IEventBinding<T> binding) => bindings.Remove(binding);

    public static void Raise(T @event)
    {
        foreach (IEventBinding<T> binding in bindings)
        {
            binding.OnEvent?.Invoke(@event);
            binding.OnEventNoArgs?.Invoke();
        }
    }
}

public interface IEventBinding<T>
{
    Action<T> OnEvent { get; set; }
    Action OnEventNoArgs { get; set; }
}

public class EventBinding<T> : IEventBinding<T> where T : struct
{
    private Action<T> onEvent = _ => { };
    private Action onEventNoArgs = () => { };

    Action<T> IEventBinding<T>.OnEvent { get => onEvent; set => onEvent = value; }
    Action IEventBinding<T>.OnEventNoArgs { get => onEventNoArgs; set => onEventNoArgs = value; }

    public EventBinding(Action<T> onEvent) => this.onEvent = onEvent;
    public EventBinding(Action onEventNoArgs) => this.onEventNoArgs = onEventNoArgs;

    public void Add(Action response)       => onEventNoArgs += response;
    public void Remove(Action response)    => onEventNoArgs -= response;
    public void Add(Action<T> response)    => onEvent += response;
    public void Remove(Action<T> response) => onEvent -= response;
}

export type AuthResult<T> =
  | { ok: true;  data: T }
  | { ok: false; error: string };

export class AuthService {
  private readonly supabase: SupabaseClient;

  constructor() {
    this.supabase = createClient(SUPABASE_URL, SUPABASE_ANON_KEY, {
      auth: { persistSession: true, autoRefreshToken: true, detectSessionInUrl: false },
    });
  }

  async signUp(email: string, password: string, username: string): Promise<AuthResult<Session>> {
    const taken = await this.isUsernameTaken(username);
    if (taken) return { ok: false, error: 'Username already taken.' };

    const { data, error } = await this.supabase.auth.signUp({
      email, password, options: { data: { username } },
    });

    if (error)          return { ok: false, error: this.formatAuthError(error) };
    if (!data.session)  return { ok: false, error: 'Awaiting email confirmation.' };

    await this.createPlayerProfile(data.session.user.id, username);
    return { ok: true, data: data.session };
  }

  async signIn(email: string, password: string): Promise<AuthResult<Session>> {
    const { data, error } = await this.supabase.auth.signInWithPassword({ email, password });
    if (error || !data.session) return { ok: false, error: this.formatAuthError(error!) };
    await this.updateLastSeen(data.session.user.id);
    return { ok: true, data: data.session };
  }

  async getProfile(userId: string): Promise<AuthResult<PlayerProfile>> {
    const { data, error } = await this.supabase
      .from('player_profiles').select('*').eq('id', userId).single();
    if (error || !data) return { ok: false, error: 'Profile not found.' };
    return { ok: true, data: data as PlayerProfile };
  }

  async refreshSession(): Promise<AuthResult<Session>> {
    const { data, error } = await this.supabase.auth.refreshSession();
    if (error || !data.session) return { ok: false, error: 'Session expired.' };
    return { ok: true, data: data.session };
  }

  async signOut(): Promise<void> { await this.supabase.auth.signOut(); }

  private async isUsernameTaken(username: string): Promise<boolean> {
    const { count } = await this.supabase
      .from('player_profiles').select('id', { count: 'exact', head: true }).ilike('username', username);
    return (count ?? 0) > 0;
  }

  private async createPlayerProfile(userId: string, username: string): Promise<void> {
    await this.supabase.from('player_profiles').insert({
      id: userId, username, display_name: username, elo_rating: 1000, total_matches: 0, wins: 0,
    });
  }

  private async updateLastSeen(userId: string): Promise<void> {
    await this.supabase.from('player_profiles')
      .update({ last_seen_at: new Date().toISOString() }).eq('id', userId);
  }

  private formatAuthError(error: AuthError): string {
    const map: Record<string, string> = {
      'Invalid login credentials': 'Incorrect email or password.',
      'Email not confirmed':        'Please confirm your email first.',
      'User already registered':    'An account with this email already exists.',
    };
    return map[error.message] ?? error.message;
  }
}

const CACHE_TTL_MS = 30_000;

class LeaderboardCache {
  private entries = new Map<string, { data: LeaderboardEntry[]; ts: number }>();

  get(key: string): LeaderboardEntry[] | null {
    const cached = this.entries.get(key);
    if (!cached) return null;
    if (Date.now() - cached.ts > CACHE_TTL_MS) { this.entries.delete(key); return null; }
    return cached.data;
  }

  set(key: string, data: LeaderboardEntry[]): void {
    this.entries.set(key, { data, ts: Date.now() });
  }

  invalidate(key: string): void {
    for (const k of this.entries.keys())
      if (k.startsWith(key)) this.entries.delete(k);
  }
}

const cache = new LeaderboardCache();

export async function getLeaderboard(
  gameMode: string, limit = 100, offset = 0
): Promise<LeaderboardEntry[]> {
  const cacheKey = `${gameMode}:${limit}:${offset}`;
  const cached = cache.get(cacheKey);
  if (cached) return cached;

  const { data, error } = await supabase
    .from('leaderboard_view')
    .select('rank,player_id,username,display_name,score,elo_rating,wins,total_matches')
    .eq('game_mode', gameMode)
    .order('score', { ascending: false })
    .range(offset, offset + limit - 1);

  if (error) throw new Error(`Leaderboard fetch failed: ${error.message}`);
  const entries = (data ?? []) as LeaderboardEntry[];
  cache.set(cacheKey, entries);
  return entries;
}

export async function submitScore(payload: SubmitScorePayload): Promise<void> {
  const { playerId, gameMode, score, metadata } = payload;

  const { data: existing } = await supabase
    .from('scores').select('id,score').eq('player_id', playerId).eq('game_mode', gameMode).single();

  if (existing && existing.score >= score) return;

  const upsertData = {
    player_id: playerId, game_mode: gameMode, score,
    metadata: metadata ?? {}, submitted_at: new Date().toISOString(),
  };

  const { error } = existing
    ? await supabase.from('scores').update(upsertData).eq('id', existing.id)
    : await supabase.from('scores').insert(upsertData);

  if (error) throw new Error(`Score submit failed: ${error.message}`);
  cache.invalidate(gameMode);
}

export async function getPlayerRank(playerId: string, gameMode: string): Promise<number | null> {
  const { data, error } = await supabase
    .from('leaderboard_view').select('rank')
    .eq('player_id', playerId).eq('game_mode', gameMode).single();
  if (error || !data) return null;
  return data.rank as number;
}

export function subscribeToLeaderboard(
  gameMode: string,
  onUpdate: (entries: LeaderboardEntry[]) => void
): RealtimeChannel {
  return supabase
    .channel(`leaderboard:${gameMode}`)
    .on('postgres_changes',
      { event: '*', schema: 'public', table: 'scores', filter: `game_mode=eq.${gameMode}` },
      async () => {
        cache.invalidate(gameMode);
        const fresh = await getLeaderboard(gameMode, 100, 0);
        onUpdate(fresh);
      })
    .subscribe();
}

export async function getInventory(playerId: string): Promise<InventoryItem[]> {
  const { data, error } = await supabase
    .from('inventory_view').select('*').eq('player_id', playerId)
    .order('acquired_at', { ascending: false });
  if (error) throw new Error(`Inventory fetch failed: ${error.message}`);
  return (data ?? []) as InventoryItem[];
}

export async function grantItem(payload: GrantItemPayload): Promise<InventoryItem> {
  const { playerId, itemId, quantity = 1, source = 'system' } = payload;

  const { data: itemDef, error: defErr } = await supabase
    .from('item_definitions').select('id,is_stackable,max_stack').eq('id', itemId).single();
  if (defErr || !itemDef) throw new Error(`Item definition not found: ${itemId}`);

  if (itemDef.is_stackable) {
    const { data: existing } = await supabase
      .from('player_inventory').select('instance_id,quantity')
      .eq('player_id', playerId).eq('item_id', itemId).single();

    if (existing) {
      const newQty = Math.min(existing.quantity + quantity, itemDef.max_stack);
      const { data, error } = await supabase
        .from('player_inventory').update({ quantity: newQty })
        .eq('instance_id', existing.instance_id).select().single();
      if (error) throw new Error(`Stack update failed: ${error.message}`);
      return data as InventoryItem;
    }
  }

  const { data, error } = await supabase
    .from('player_inventory')
    .insert({ player_id: playerId, item_id: itemId, quantity, source, acquired_at: new Date().toISOString() })
    .select().single();
  if (error) throw new Error(`Item grant failed: ${error.message}`);
  return data as InventoryItem;
}

export async function consumeItem(playerId: string, instanceId: string, amount = 1): Promise<void> {
  const { data: item, error: fetchErr } = await supabase
    .from('player_inventory').select('quantity')
    .eq('player_id', playerId).eq('instance_id', instanceId).single();
  if (fetchErr || !item) throw new Error('Item not found in inventory.');
  if (item.quantity < amount) throw new Error('Insufficient quantity.');

  const newQty = item.quantity - amount;
  if (newQty === 0) {
    const { error } = await supabase.from('player_inventory').delete().eq('instance_id', instanceId);
    if (error) throw new Error(`Delete failed: ${error.message}`);
  } else {
    const { error } = await supabase.from('player_inventory')
      .update({ quantity: newQty }).eq('instance_id', instanceId);
    if (error) throw new Error(`Quantity update failed: ${error.message}`);
  }
}

export async function setEquipped(playerId: string, instanceId: string, equipped: boolean): Promise<void> {
  const { error } = await supabase
    .from('player_inventory').update({ is_equipped: equipped })
    .eq('player_id', playerId).eq('instance_id', instanceId);
  if (error) throw new Error(`Equip toggle failed: ${error.message}`);
}

const CHARS = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789#@&$';

function scramble(el, finalText, durationMs) {
  const totalFrames = 50;
  let frame = 0;
  const id = setInterval(function() {
    el.textContent = finalText.split('').map(function(ch, i) {
      if (ch === ' ' || ch === '.') return ch;
      if (frame / totalFrames > i / finalText.length) return ch;
      return CHARS[Math.floor(Math.random() * CHARS.length)];
    }).join('');
    if (++frame > totalFrames) { clearInterval(id); el.textContent = finalText; }
  }, durationMs / totalFrames);
}

function initParticles() {
  const canvas = document.getElementById('particle-canvas');
  const ctx = canvas.getContext('2d');
  let W, H, particles;
  const COUNT = 90;
  const CONNECT = 130;
  const A = [180, 255, 0];

  function resize() { W = canvas.width = innerWidth; H = canvas.height = innerHeight; }

  function makeP() {
    const angle = Math.random() * Math.PI * 2;
    const spd = 0.12 + Math.random() * 0.22;
    return { x: Math.random()*W, y: Math.random()*H,
             vx: Math.cos(angle)*spd, vy: Math.sin(angle)*spd,
             r: 0.6 + Math.random()*1.6, op: 0.1 + Math.random()*0.35 };
  }

  function init() { particles = Array.from({length:COUNT}, makeP); }

  function draw() {
    ctx.clearRect(0,0,W,H);
    for (let i = 0; i < particles.length-1; i++) {
      for (let j = i+1; j < particles.length; j++) {
        const dx = particles[i].x - particles[j].x;
        const dy = particles[i].y - particles[j].y;
        const d = Math.sqrt(dx*dx+dy*dy);
        if (d > CONNECT) continue;
        const al = (1-d/CONNECT)*0.07;
        ctx.beginPath();
        ctx.moveTo(particles[i].x, particles[i].y);
        ctx.lineTo(particles[j].x, particles[j].y);
        ctx.strokeStyle = `rgba(${A},${al})`;
        ctx.lineWidth = 0.5;
        ctx.stroke();
      }
    }
    particles.forEach(function(p) {
      p.x += p.vx; p.y += p.vy;
      if (p.x < -8) p.x = W+8; if (p.x > W+8) p.x = -8;
      if (p.y < -8) p.y = H+8; if (p.y > H+8) p.y = -8;
      ctx.beginPath();
      ctx.arc(p.x, p.y, p.r, 0, Math.PI*2);
      ctx.fillStyle = `rgba(${A},${p.op})`;
      ctx.fill();
    });
    requestAnimationFrame(draw);
  }

  addEventListener('resize', function() { resize(); init(); });
  resize(); init(); draw();
}

function initScrollAnimations() {
  const obs = new IntersectionObserver(function(entries) {
    entries.forEach(function(e) {
      if (!e.isIntersecting) return;
      e.target.classList.add('revealed');
      obs.unobserve(e.target);
    });
  }, { threshold: 0.08, rootMargin: '0px 0px -20px 0px' });

  document.querySelectorAll('.skill-block, .gm-item, .vr-card, .exp-item, .reveal')
    .forEach(function(el) { obs.observe(el); });
}

function initSkillBars() {
  const barObs = new IntersectionObserver(function(entries) {
    entries.forEach(function(e) {
      if (!e.isIntersecting) return;
      const fill = e.target.querySelector('.skill-bar-fill');
      if (fill) fill.style.width = fill.dataset.pct + '%';
      barObs.unobserve(e.target);
    });
  }, { threshold: 0.3 });
  document.querySelectorAll('.skill-block').forEach(function(el) { barObs.observe(el); });
}

document.addEventListener('DOMContentLoaded', function() {
  initParticles();
  initScrollAnimations();
  initSkillBars();
});
